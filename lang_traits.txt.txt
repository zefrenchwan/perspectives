=======================
== TRAITS DEFINITION ==
=======================

TYPE => string | int | bool | time | float

SCRIPT_FILE => (TRAIT_CREATION|TRAIT_DELETION)+
SHELL_SESSION => (TRAIT_ORDER \n)*

TRAIT_ORDER => TRAIT_CREATION | TRAIT_LIST | TRAIT DESC | TRAIT_DELETION

TRAIT_DELETION => delete trait TRAIT_NAME \n

TRAIT_DESC => describe trait TRAIT_NAME \n 

TRAIT_LIST => list traits \n 

TRAIT_CREATION => trait TRAIT_NAME (TRAIT_EXTENSION){0,1}\n TRAIT_ACTIONS \n
TRAIT_EXTENSION => extends TRAIT_NAME (and TRAIT_NAME)*
TRAIT_ACTIONS => (\t with ATTRIBUTE_DECLARATION \n)+
ATTRIBUTE_DECLARATION => NAME : TYPE
